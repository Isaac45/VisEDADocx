Hyperspectral EDA

Documentation for single-cube and dataset-level hyperspectral workflows.

Single cube: HyperspectralEDA

import numpy as np
from viseda import HyperspectralEDA

wavelengths = np.linspace(400, 2500, 200)
eda = HyperspectralEDA(wavelengths=wavelengths)
eda.load("scene.hdr")
summary = eda.summary()
wl, sig = eda.spectral_signature(row=100, col=200)
scores, variance_ratio = eda.pca(n_components=10)
ndvi = eda.compute_ndvi(nir_band=140, red_band=60)
eda.plot(save_path="hyper_dashboard.png")

Dataset-level: HyperspectralDatasetEDA

from viseda import HyperspectralDatasetEDA

eda = HyperspectralDatasetEDA(wavelengths=wavelengths)
eda.load("path/to/hyperspectral_dataset/", label_from_parent=True)
summary = eda.summary()
eda.plot(save_path="hyper_dataset_dashboard.png")
