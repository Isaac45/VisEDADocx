Point Cloud EDA

Documentation for single-cloud and dataset-level point cloud workflows.

Single cloud: PointCloudEDA

from viseda import PointCloudEDA

eda = PointCloudEDA(max_points=500_000)
eda.load("scan.las")
summary = eda.summary()
eda.plot(save_path="cloud_dashboard.png")
eda.plot_3d(color_by="z")

Dataset-level: PointCloudDatasetEDA

from viseda import PointCloudDatasetEDA

eda = PointCloudDatasetEDA(max_points_per_cloud=200_000)
eda.load("path/to/clouds/", label_from_parent=True)
summary = eda.summary()
eda.plot(save_path="cloud_dataset_dashboard.png")
