Quickstart

Minimal examples for the public VisEDA API.

Image dataset

from viseda import ImageEDA

eda = ImageEDA(verbose=True)
eda.load("path/to/dataset/", label_from_parent=True)
summary = eda.summary()
print(summary["quality"]["brightness"])
print(eda.normalization_stats())
eda.plot_colour(save_path="colour_dashboard.png")
eda.plot_samples(n=25, cols=5, save_path="samples.png")
eda.report("image_report.html")

Hyperspectral cube

import numpy as np
from viseda import HyperspectralEDA

wavelengths = np.linspace(400, 2500, 200)
eda = HyperspectralEDA(wavelengths=wavelengths)
eda.load("scene.hdr")
summary = eda.summary()
wl, signature = eda.spectral_signature(row=100, col=200)
ndvi = eda.compute_ndvi(nir_band=140, red_band=60)
eda.plot(save_path="hyperspectral_dashboard.png")

Point cloud

from viseda import PointCloudEDA

eda = PointCloudEDA(max_points=500_000)
eda.load("scan.las")
summary = eda.summary()
eda.plot(save_path="pointcloud_dashboard.png")
eda.plot_3d(color_by="z")
