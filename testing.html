Testing File

A practical testimg.py script for exercising the current ImageEDA implementation without editing eda.py.

The uploaded library version may raise a radar-axis error when running plot() or plot_quality(). This test file focuses on functions that currently run safely from outside the library.
from pathlib import Path
import numpy as np
from viseda import ImageEDA

DATASET_PATH = "C:/Users/Ike/Desktop/testvis/"
OUTPUT_DIR = Path("image_eda_outputs")
OUTPUT_DIR.mkdir(exist_ok=True)

eda = ImageEDA(
    verbose=True,
    max_images=None,
    n_colors=8,
    phash_threshold=10,
    blur_threshold=50.0,
    compute_glcm=True,
    compute_freq=True,
)

eda.load(DATASET_PATH, label_from_parent=True, recursive=True)
summary = eda.summary()

print("
===== INVENTORY =====")
print(summary["inventory"])

print("
===== SPATIAL =====")
print(summary["spatial"])

print("
===== QUALITY: BRIGHTNESS =====")
print(summary["quality"]["brightness"])

print("
===== QUALITY: SHARPNESS =====")
print(summary["quality"]["sharpness"])

print("
===== COLOUR =====")
print(summary["colour"])

print("
===== TEXTURE =====")
print(summary["texture"])

print("
===== FREQUENCY =====")
print(summary["frequency"])

print("
===== DUPLICATES =====")
print(summary["duplicates"])

print("
===== LABELS =====")
print(summary["labels"])

print("
===== NORMALIZATION STATS =====")
print(eda.normalization_stats())

# Safe visualisations
eda.plot_colour(save_path=str(OUTPUT_DIR / "colour_dashboard.png"))
eda.plot_texture(save_path=str(OUTPUT_DIR / "texture_dashboard.png"))
eda.plot_samples(n=25, cols=5, save_path=str(OUTPUT_DIR / "sample_images.png"))
eda.plot_class_samples(n_per_class=5, save_path=str(OUTPUT_DIR / "class_samples.png"))
eda.plot_average_image(target_size=(224, 224), save_path=str(OUTPUT_DIR / "average_image.png"))
eda.plot_channel_correlation(max_pixels=50_000, save_path=str(OUTPUT_DIR / "channel_correlation.png"))
eda.plot_duplicates(mode="exact", max_groups=5, save_path=str(OUTPUT_DIR / "exact_duplicates.png"))
eda.plot_duplicates(mode="near", max_groups=5, save_path=str(OUTPUT_DIR / "near_duplicates.png"))
eda.report(str(OUTPUT_DIR / "image_eda_report.html"))

# Test NumPy array loading separately
arrays = [np.random.randint(0, 255, (256, 256, 3), dtype=np.uint8) for _ in range(10)]
array_labels = ["no damage", "minor", "moderate", "heavy", "no damage", "minor", "moderate", "heavy", "no damage", "minor"]

eda_arrays = ImageEDA(verbose=True)
eda_arrays.load_arrays(arrays, labels=array_labels)
array_summary = eda_arrays.summary()

print("
===== ARRAY TEST SUMMARY =====")
print(array_summary["inventory"])
print(array_summary["quality"]["brightness"])
print(array_summary["labels"])

eda_arrays.plot_texture(save_path=str(OUTPUT_DIR / "array_texture_dashboard.png"))
eda_arrays.report(str(OUTPUT_DIR / "array_report.html"))

print("
All safe ImageEDA tests completed successfully.")
