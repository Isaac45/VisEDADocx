Safe ImageEDA script

This version exercises the stable calls without modifying eda.py.

from pathlib import Path
import numpy as np
from viseda import ImageEDA

DATASET_PATH = "C:/Users/Ike/Desktop/testvis/"
OUTPUT_DIR = Path("image_eda_outputs")
OUTPUT_DIR.mkdir(exist_ok=True)

eda = ImageEDA(verbose=True, compute_glcm=True, compute_freq=True)
eda.load(DATASET_PATH, label_from_parent=True, recursive=True)

summary = eda.summary()
print(summary["inventory"])
print(summary["quality"]["brightness"])
print(summary["colour"])
print(summary["texture"])
print(summary["frequency"])
print(eda.normalization_stats())

eda.plot_colour(save_path=str(OUTPUT_DIR / "colour_dashboard.png"))
eda.plot_texture(save_path=str(OUTPUT_DIR / "texture_dashboard.png"))
eda.plot_samples(n=25, cols=5, save_path=str(OUTPUT_DIR / "sample_images.png"))
eda.plot_class_samples(n_per_class=5, save_path=str(OUTPUT_DIR / "class_samples.png"))
eda.plot_average_image(save_path=str(OUTPUT_DIR / "average_image.png"))
eda.plot_channel_correlation(save_path=str(OUTPUT_DIR / "channel_correlation.png"))
eda.plot_duplicates(mode="exact", save_path=str(OUTPUT_DIR / "exact_duplicates.png"))
eda.plot_duplicates(mode="near", save_path=str(OUTPUT_DIR / "near_duplicates.png"))
eda.report(str(OUTPUT_DIR / "image_eda_report.html"))

arrays = [np.random.randint(0, 255, (256, 256, 3), dtype=np.uint8) for _ in range(10)]
labels = ["no damage", "minor", "moderate", "heavy", "no damage", "minor", "moderate", "heavy", "no damage", "minor"]
eda_arrays = ImageEDA(verbose=True)
eda_arrays.load_arrays(arrays, labels=labels)
print(eda_arrays.summary()["quality"]["brightness"])
eda_arrays.plot_texture(save_path=str(OUTPUT_DIR / "array_texture_dashboard.png"))
eda_arrays.report(str(OUTPUT_DIR / "array_report.html"))
